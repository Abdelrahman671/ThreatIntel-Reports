  /**
   * Finds and returns the submit button for the form containing the given element.
   * @param {HTMLElement} element - The HTML element inside the form.
   * @returns {HTMLButtonElement | null} The submit button of the form, or null if not found.
   */
  function findFormSubmitButton(element) {
    // First, ensure the element is valid
    if (!(element instanceof HTMLElement)) {
      console.error("Invalid element provided");
      return null;
    }

    // Find the closest form ancestor
    const neverbounceForm = element.closest("form");

    // If no form is found, return null
    if (!neverbounceForm) {
      console.warn("No form found for the given element");
      return null;
    }

    // Look for a submit button within the form
    const submitButton = neverbounceForm.querySelector('button[type="submit"], input[type="submit"]');

    // Return the submit button or null if not found
    return submitButton || null;
  }

  /**
   * Hook into field registration events
   * The nb:registered event is fired on the body every time
   * _nb.fields.registerListener is called and during page load for

   *
   * All `nb:` prefixed events contain a `detail` object
   * with the following params:
   *   `id` - a reference ID for the field & form
   *   `result` - a result object
   *   `error` - error object from request
   */
  document.querySelector("body").addEventListener("nb:registered", function (event) {
    // Get field using id from registered event
    let field = document.querySelector('[data-nb-id="' + event.detail.id + '"]');

    // Handle clear events (input has changed or an API error was returned)
    field.addEventListener("nb:clear", function (e) {
      // Check for errors
      if (e.detail.result && e.detail.result.isError()) {
        if (e.detail.result.isThrottled()) {
          // Do stuff when the verification is throttled
        }

        // Do stuff when other API errors occur
        // - Our recommendation is to hide any loaders and treat these emails the same way you would treat an Unknown email
        const submitButton = findFormSubmitButton(e.target);
        if (submitButton) {
          submitButton.disabled = "";
        }
        e.target.setCustomValidity("");
      } else {
        if (!e.target.value) {
          e.target.setCustomValidity("");
        } else {
          e.target.setCustomValidity("Please provide your company email. Free domain emails do not apply.");
        }
      }
      // Do stuff when input changes, (e.g. hide loader)
    });

    // Handle loading status (API request has been made)
    field.addEventListener("nb:loading", function (e) {
      // Do stuff while waiting on API response
      const submitButton = findFormSubmitButton(e.target);
      if (submitButton) {
        submitButton.disabled = "disabled";
      }

      var id = e.detail.id;

      // Get result object
      var result = e.detail.result;

<script>
  document.addEventListener("DOMContentLoaded", (event) => {
    document.querySelectorAll(".huntress-form-group input").forEach((field) => {
      const label = field.closest(".huntress-form-group").querySelector("label");

      // Function to add the shrink-label class
      const addShrinkLabel = () => {
        if (label) {
          label.classList.add("shrink-label");
        }
      };

      // Function to remove the shrink-label class if input is empty
      const removeShrinkLabelIfEmpty = () => {
        if (label && field.value.trim() === "") {
          label.classList.remove("shrink-label");
        }
      };

      // Add class on focus
      field.addEventListener("focus", addShrinkLabel);

      // Remove class on blur if input is empty
      field.addEventListener("blur", removeShrinkLabelIfEmpty);

      // Handle input event as well
      field.addEventListener("input", removeShrinkLabelIfEmpty);
    });
  });
</script>

      const emailValue = emailInput.value;

    async function hubSpotFormSubmitHandler(eventForm) {
      if (eventForm.getAttribute("data-hubspot-form-id")) {
        var url = `https://api.hsforms.com/submissions/v3/integration/submit/3911692/${eventForm.getAttribute("data-hubspot-form-id")}`;

        var formData = new FormData(eventForm);

        let fields = [];
        let context = {};

        const contextFields = ["hutk", "ipAddress", "pageUri", "pageId", "pageName"];
        const companyFields = ["name"];
        const companyPrefix = "0-2/";
        const contractFields = ["title"];
        const contractPrefix = "2-14533744/";

        Array.from(formData.entries()).forEach((entry) => {
          const elements = document.getElementsByName(entry[0]);
          if (elements) {
            if (elements[0]) {
              const fieldName = elements[0].getAttribute("data-hubspot-fieldname") ?? elements[0].getAttribute("data-wfhsfieldname") ?? entry[0];
              if (elements[0]?.type == "hidden" && contextFields.includes(fieldName)) {
                if (!!entry[1]) {
                  context[fieldName] = entry[1];
                } else {
                  if (entry[0] == "pageUri") {
                    context.pageUri = window.location.href;
                  }
                }
              } else {
                let field = {
                  name: fieldName,
                  value: entry[1],
                };
                if (companyFields.includes(fieldName)) field.name = companyPrefix + fieldName;
                if (contractFields.includes(fieldName)) field.name = contractPrefix + fieldName;
                fields.push(field);
              }
            }
          }
        });

        const data = { fields: fields, context: context };

        const headers = new Headers();
        headers.append("Content-Type", "application/json");

        const response = await fetch(url, {
          method: "POST",
          body: JSON.stringify(data),
          headers: headers,
        });
        if (response?.status && response.status < 400) {
          return response;
          // setSuccessState(eventForm, successElements, failElements)
          // setSubmitButtonValue(submitButton, submitButtonValue)
        } else {
          console.error("Form submit error", response);
          throw new Error("Form submit Error");
          // setSubmitButtonValue(submitButton, submitButtonValue)
          // setFailedState(eventForm, successElements, failElements)
        }
      } else {
        console.log(`No HubSpot form id defined for ${eventForm.id}`);
        throw new Error(`No HubSpot form id defined for ${eventForm.id}`);
      }
    }

    async function formSubmitListener(e, handlePreFormSubmit, handleFormSubmit, handlePostFormSubmitSuccess, handlePostFormSubmitFailure) {

          alert("Please validate the Captcha test");
          return;

        let formBlock = eventForm.closest(".form-block");
        let successElements = formBlock.querySelectorAll('[class*="w-form-done"]');
        let failElements = formBlock.querySelectorAll('[class*="w-form-fail"]');

        resetFormState(successElements, failElements);

        setSubmitButtonValue(submitButton, submitButtonWait, true);
        try {
          if (typeof handleFormSubmit !== "function") {
            throw new Error("handleFormSubmit is not a function");
          }

          hubSpotPreFormSubmitHandler(eventForm);

          if (typeof handlePreFormSubmit === "function") {
            try {
              handlePreFormSubmit(eventForm);
            } catch (error) {
              console.log("handlePreFormSubmit Error", error);
            }
          } else {
            console.log("No pre form submit handler present");
          }

          const result = await handleFormSubmit(eventForm);
          setSuccessState(eventForm, successElements, failElements);
          setSubmitButtonValue(submitButton, submitButtonValue);

          if (typeof handlePostFormSubmitSuccess === "function") {
            try {
              handlePostFormSubmitSuccess(eventForm);
            } catch (error) {
              console.log("handlePostFormSubmitSuccess Error", error);
            }
          } else {
            console.log("No post form success handler present");
          }
        } catch (error) {
          console.error("formSubmissionListener error", error);
          if (typeof handlePostFormSubmitFailure === "function") {
            try {
              handlePostFormSubmitFailure(eventForm);
            } catch (error) {
              console.log("handlePostFormSubmitFailure Error", error);
            }
          } else {
            console.log("No post form failure handler present");
          }
          setSubmitButtonValue(submitButton, submitButtonValue);
          setFailedState(eventForm, successElements, failElements);
        }
      } catch (error) {
        console.log("formSubmitListener error", error);
      }
    }

    const submitButtons = document.querySelectorAll(".form-block form input[type='submit']");
