  function validateEmailAddress(emailInput, ignoreNB, ignoreFreeDomains) {
    const email = emailInput.value;

    let checkResult = false;
    if (email.match(burnerDomainsRegex) || (!ignoreFreeDomains && email.match(freeDomainsRegex)) || (!ignoreNB && !nbResponseResult(emailInput))) {
      if (ignoreFreeDomains) {
        emailInput.setCustomValidity("Please provide a valid email address.");
      } else {
        emailInput.setCustomValidity("Please provide your company email. Free domain emails do not apply.");
      }
    } else {
      emailInput.setCustomValidity("");
      checkResult = true;
    }
    return checkResult;
  }

    } else {
      return false;
    }
  }

  async function submitClickListener(e, handleSubmitClick) {
    try {
      let eventForm = e.currentTarget.closest("form");

        if (typeof handleSubmitClick === "function") {

      }

        });
      }

      return eventForm.reportValidity();

      console.error("submitClickListener error", error);
      return false;
    }
  }

  function hubSpotPreFormSubmitHandler(eventForm) {

  async function hubSpotFormSubmitHandler(eventForm) {
    if (eventForm.getAttribute("data-hubspot-form-id")) {
      var url = `https://api.hsforms.com/submissions/v3/integration/submit/3911692/${eventForm.getAttribute("data-hubspot-form-id")}`;

      var formData = new FormData(eventForm);

      let fields = [];
      let context = {};

      const contextFields = ["hutk", "ipAddress", "pageUri", "pageId", "pageName"];
      const companyFields = ["name"];
      const companyPrefix = "0-2/";
      const contractFields = ["title"];
      const contractPrefix = "2-14533744/";

      Array.from(formData.entries()).forEach((entry) => {
        const elements = document.getElementsByName(entry[0]);
        if (elements) {
          if (elements[0]) {
            const fieldName = elements[0].getAttribute("data-hubspot-fieldname") ?? elements[0].getAttribute("data-wfhsfieldname") ?? entry[0];
            if (elements[0]?.type !== "password") {
              if (elements[0]?.type == "hidden" && contextFields.includes(fieldName)) {
                if (!!entry[1]) {
                  context[fieldName] = entry[1];
                } else {
                  if (entry[0] == "pageUri") {
                    context.pageUri = window.location.href;
                  }
                }
              } else {
                let field = {
                  name: fieldName,
                  value: entry[1],
                };
                if (companyFields.includes(fieldName)) field.name = companyPrefix + fieldName;
                if (contractFields.includes(fieldName)) field.name = contractPrefix + fieldName;
                fields.push(field);
              }
            }
          }
        }
      });

      const data = { fields: fields, context: context };

      const headers = new Headers();
      headers.append("Content-Type", "application/json");

      const response = await fetch(url, {
        method: "POST",
        body: JSON.stringify(data),
        headers: headers,
      });
      if (response?.status && response.status < 400) {
        return response;
      } else {
        console.error("Form submit error", response);
        throw new Error("Form submit Error");
      }
    } else {
      console.log(`No HubSpot form id defined for ${eventForm.id}`);
      throw new Error(`No HubSpot form id defined for ${eventForm.id}`);
    }
  }

  async function formSubmitListener(e, handlePreFormSubmit, handleFormSubmit, handlePostFormSubmitSuccess, handlePostFormSubmitFailure) {
    try {
      e.preventDefault();
      let eventForm = e.currentTarget;

      const submitButton = eventForm.querySelector("input[type='submit']");
      const submitButtonValue = submitButton.value;
      const submitButtonWait = submitButton.getAttribute("data-wait");

      if (!isValidRecaptcha(eventForm)) {
        alert("Please validate the Captcha test");
        return;
      }

      resetFormState(eventForm);

      setSubmitButtonValue(submitButton, submitButtonWait, true);
      try {
        if (typeof handleFormSubmit !== "function") {
          throw new Error("handleFormSubmit is not a function");
        }

        hubSpotPreFormSubmitHandler(eventForm);

        if (typeof handlePreFormSubmit === "function") {
          try {
            const preFormSubmitResponse = handlePreFormSubmit(eventForm);
            if (!preFormSubmitResponse) {
              setSubmitButtonValue(submitButton, submitButtonValue);
              return false;
            }
          } catch (error) {
            console.log("handlePreFormSubmit Error", error);
          }
        }

        const result = await handleFormSubmit(eventForm);
        if (typeof handlePostFormSubmitSuccess === "function") {
          try {
            await handlePostFormSubmitSuccess(eventForm);
          } catch (error) {
            console.log("handlePostFormSubmitSuccess Error", error);
            setSubmitButtonValue(submitButton, submitButtonValue);
            return false;
          }
        }
        setSuccessState(eventForm);
        setCorrelationId(eventForm);
        setSubmitButtonValue(submitButton, submitButtonValue);
      } catch (error) {
        console.error("formSubmissionListener error", error);
        if (typeof handlePostFormSubmitFailure === "function") {
          try {
            await handlePostFormSubmitFailure(eventForm);
          } catch (error) {
            console.log("handlePostFormSubmitFailure Error", error);
          }
        } else {
          setFailedState(eventForm);
        }
        // setCorrelationId(eventForm);
        setSubmitButtonValue(submitButton, submitButtonValue);
        return false;
      }
    } catch (error) {
      console.log("formSubmitListener error", error);
    }
  }

activity across several organizations in Canada, with similar infrastructure and TTPs used that can be associated with the APT group known as RedCurl (aka <a This activity goes back as far as November 2023 in the hosts observed by Huntress.</p><p>Historically, RedCurl has attacked with the intent of cyberespionage, seeking to remain undetected for long periods of time to access and gather data from emails, corporate documents, and other confidential files from their victimsâthey donât usually encrypt systems, steal money, or even demand a ransom for the data they steal from the victim while in their network. RedCurl has targeted large wholesale retailers in the past as well as companies in the finance, tourism, insurance, construction, and consulting industries.</p><p>Here, we aim to demonstrate the interesting and unique tactics utilized by RedCurl and highlight similarities between the attacks we observed and previously documented attacks. Weâll also provide some insights and guidance on what can be learned from these incidents and how to detect similar techniques that RedCurl or any other malware or adversaries could use in the future.Â Â </p><p>â</p><h2><strong>What Did We Find?</strong></h2><p>The Huntress Security Operations Center (SOC) received an alert on a host that had recently installed the Huntress agent: a 7zip binary executed from a suspicious location, which triggered an alert. From there, we found related persistence on the host. We discovered several scheduled tasks that were used to execute the<strong> Windows Program Compatibility Assistant ([.highlight]pcalua.exe[.highlight])</strong>, which then executed a malicious binary. We later identified other tasks that used <strong>[.highlight]pcalua.exe[.highlight]</strong> to run Python scripts. The Python script used in the command (<strong>[.highlight]cl.py[.highlight]</strong>) is exactly the same as the <a href="https://github.com/klsecservices/rpivot/blob/master/client.py" target="_blank" rel="nofollow">RPivot</a> (proxy tool) script called<strong> [.highlight]client.py[.highlight]</strong>. This is used to start a connection to a proxy, presumably to connect back to a remote server for Command and Control. We also found some evidence suggesting that data may have been collected, archived, and exfiltrated to cloud storage.</p><p>Upon investigation of the initial host, we identified the same tradecraft used on additional hosts and at various other organizations. We uncovered three intrusions across three different organizations that were all fully or partially located in Canada. The activity observed in these attacks all match very closely with known behaviors from RedCurl, and much of the behavior is fairly unique and distinct from other similar procedures from more common malware or other groups. RedCurl continues to change and modify some of their procedures, adding new wrinkles here and there to make detection more difficult.</p><p>â</p><h2><strong>RedCurl Tradecraft</strong></h2><p>While the intent and methods used in the cases we observed match closely with known RedCurl behavior, the specific techniques and infection chain used (as we were able to observe) differed slightly from previous reported cases.<br/><br/>While we werenât able to obtain all the files used (many were deleted by the malware very quickly, and we didnât observe many of the cases all the way from the initial infection), we did find some executables and batch scripts that we were able to analyze. This way, they can remain undetected and gather more information without tripping detections and alerting anyone to their presence. Many of the commands we observed could also be found during perfectly normal activity from legitimate processes. Detecting this type of activity can be difficult without creative ways to hunt for new and interesting behaviors in the network. The more layers of defense you have in place, the more likely it is that one layer will surface some interesting activity and potentially uncover a serious attack from a highly motivated threat actor who may be interested in the data that can be found in your network.</p><p>â</p><h2>IOCs</h2><h3>Files<br/></h3><div class="w-embed"><!DOCTYPE html>
