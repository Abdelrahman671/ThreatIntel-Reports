    function makeYearsSelect( years, isAddYearPickerHint )
    {
        var options = isAddYearPickerHint
              ? '<option value="">' + str_choose_year + '</option>'
              : '',
            count;

        for ( var count = 0; count < years.length; count++ ) {
            options += (
                '<option value="'+ years[count] +'">' +
                    years[count] +
                '</option>'
            );
        }

        return { tpl: (
            '<select class="Input space_1of1" id="newsYearsSelect" onchange="updateMonthsSelect(this.value, false, true)">' +
                options +
            '</select>'
        ) }
    }

    function makeMonthsSelect( year, isAddMonthPickerHint )
    {
        var months_year = data[year], // [{"01":"22"},{"02":"36"}]
            values = {},
            options = isAddMonthPickerHint
              ? '<option value="0" data-month="00">' + str_choose_month + '</option>'
              : '',
            counts = [],
            count,
            news;

        if ( months_year ) {

          for ( count = 0; count < months_year.length; count++ ) {
              news = getInfoFromYearObj( months_year[count] );
              values[news.id] = count;
              counts.push( news.id );
              options += (
                  '<option value="' + news.count + '" data-month="' + news.id + '">' +
                      news.month +
                  '</option>'
              );
          }

          values.max = Math.max.apply( null, counts );
          values.min = Math.min.apply( null, counts );

        }

        return { options: values, tpl: (
            '<select class="Input space_1of1" id="newsMonthsSelect" onchange="updatePage(this,' + year + ')">' +
                options +
            '</select>'
        ) }
    }

    function getInfoFromYearObj( obj )
    {
        for( var key in obj ) return {
            'month': months[ parseInt( key, 10 ) ],
            'count': obj[ key ],
            'id': key
        };
    }

          link: link,
          count: count,
          year: year,
          month: month
        }
