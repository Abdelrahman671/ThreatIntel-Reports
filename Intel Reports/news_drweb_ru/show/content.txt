    function makeMonthsSelect( year, isAddMonthPickerHint )
    {
        var months_year = data[year], // [{"01":"22"},{"02":"36"}]
            values = {},
            options = isAddMonthPickerHint
              ? '<option value="0" data-month="00">' + str_choose_month + '</option>'
              : '',

            count,
            news;

        if ( months_year ) {

          for ( count = 0; count < months_year.length; count++ ) {
              news = getInfoFromYearObj( months_year[count] );
              values[news.id] = count;
              counts.push( news.id );
              options += (
                  '<option value="' + news.count + '" data-month="' + news.id + '">' +
                      news.month +
                  '</option>'
              );
          }

          values.max = Math.max.apply( null, counts );
          values.min = Math.min.apply( null, counts );
