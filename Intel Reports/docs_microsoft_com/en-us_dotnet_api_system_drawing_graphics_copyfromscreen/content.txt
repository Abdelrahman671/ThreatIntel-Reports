  <h3>Examples</h3>
    	<p>The following code example demonstrates how to use the <a href="system.drawing.graphics.copyfromscreen?view=netframework-4.8" class="no-loc" data-linktype="relative-path">CopyFromScreen</a> for printing a copy of the current form.</p>
<pre><code class="lang-csharp">using System;
using System.Windows.Forms;
using System.Drawing;
using System.Drawing.Printing;

public class Form1 :
    Form
{
    private Button printButton = new Button();
    private PrintDocument printDocument1 = new PrintDocument();

    public Form1()
    {
        printButton.Text = "Print Form";
        printButton.Click += new EventHandler(printButton_Click);
        printDocument1.PrintPage += new PrintPageEventHandler(printDocument1_PrintPage);
        this.Controls.Add(printButton);
    }

    void printButton_Click(object sender, EventArgs e)
    {
        CaptureScreen();
        printDocument1.Print();
    }

    Bitmap memoryImage;

    private void CaptureScreen()
    {
        Graphics myGraphics = this.CreateGraphics();
        Size s = this.Size;
        memoryImage = new Bitmap(s.Width, s.Height, myGraphics);
        Graphics memoryGraphics = Graphics.FromImage(memoryImage);
        memoryGraphics.CopyFromScreen(this.Location.X, this.Location.Y, 0, 0, s);
    }

    private void printDocument1_PrintPage(System.Object sender,  
           System.Drawing.Printing.PrintPageEventArgs e)
    {
        e.Graphics.DrawImage(memoryImage, 0, 0);
    }

    public static void Main()
    {
        Application.Run(new Form1());
    }
}
</code></pre>

  <h3>Examples</h3>
    	<p>The following example demonstrates how to use the <a href="system.drawing.graphics.copyfromscreen?view=netframework-4.8" class="no-loc" data-linktype="relative-path">CopyFromScreen</a> method. To run this example, paste it into a Windows Form. Handle the form's <a href="system.windows.forms.control.paint?view=netframework-4.8#system-windows-forms-control-paint" class="no-loc" data-linktype="relative-path">Paint</a> event and call the <code>CopyPixels4</code> method from the <a href="system.windows.forms.control.paint?view=netframework-4.8#system-windows-forms-control-paint" class="no-loc" data-linktype="relative-path">Paint</a> event-handling method, passing <code>e</code> as <a href="system.windows.forms.painteventargs?view=netframework-4.8" class="no-loc" data-linktype="relative-path">PaintEventArgs</a>.</p>
<pre><code class="lang-csharp">private void CopyPixels4(PaintEventArgs e)
{
    e.Graphics.CopyFromScreen(0, 0, 20, 20, new Size(160, 160), 
        CopyPixelOperation.SourceInvert);
}
</code></pre>
<pre><code class="lang-vb">Private Sub CopyPixels4(ByVal e As PaintEventArgs) 
    e.Graphics.CopyFromScreen(0, 0, 20, 20, _
        New Size(160, 160), CopyPixelOperation.SourceInvert)
End Sub
</code></pre>


    	<h3>Remarks</h3>
    	<p>The <a href="system.drawing.graphics.copyfromscreen?view=netframework-4.8" class="no-loc" data-linktype="relative-path">CopyFromScreen</a> methods are useful for layering one image on top of another. The <code>copyPixelOperation</code> parameter allows you to specify if and how the source colors should be blended with the colors in the destination area.</p>
