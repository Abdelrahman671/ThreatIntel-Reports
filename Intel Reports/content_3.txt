import os
import sys

def delete_archived_links(root_folder):
    for dirpath, _, filenames in os.walk(root_folder):
        for filename in filenames:
            if filename == "archived_links.txt":
                file_path = os.path.join(dirpath, filename)
                os.remove(file_path)
                print(f"Deleted: {file_path}")

if __name__ == "__main__":
    if len(sys.argv) != 2:
        print("Usage: python delete_archived_links.py <path_to_directory>")
        sys.exit(1)

    root_folder = sys.argv[1]
    delete_archived_links(root_folder)
