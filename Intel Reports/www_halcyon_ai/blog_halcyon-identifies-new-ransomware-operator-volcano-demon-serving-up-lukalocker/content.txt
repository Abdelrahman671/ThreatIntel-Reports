// Function to get URL parameters
function getUrlParams() {
    const params = {};
    const searchParams = new URLSearchParams(window.location.search);
    for (const [key, value] of searchParams) {
        params[key] = value;
    }
    return params;
}

// Function to update hidden inputs across the document
function updateHiddenInputs() {
    const params = getUrlParams();
    const utmParams = ['utm_source', 'utm_medium', 'utm_campaign', 'utm_term', 'utm_content'];

    // Update UTM parameters
    utmParams.forEach(param => {
        const inputs = document.querySelectorAll(`input[type="hidden"][name="${param}"]`);
        inputs.forEach(input => {
            if (params[param]) {
                input.value = params[param];
            }
        });
    });

    // Update page title
    const titleInputs = document.querySelectorAll('input[type="hidden"][name="pageName"]');
    titleInputs.forEach(input => {
        input.value = document.title;
    });

    // Update page URL
    const urlInputs = document.querySelectorAll('input[type="hidden"][name="pageUri"]');
    urlInputs.forEach(input => {
        input.value = window.location.href;
