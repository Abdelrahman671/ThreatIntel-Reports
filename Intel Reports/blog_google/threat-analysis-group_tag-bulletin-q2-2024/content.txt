  (function() {
    var elem_ =  document.querySelector('.page-analytics-data');

    var scrubObject = {
      'event': 'scrubbedUrl',
      'url': getScrubbedUrl()
    };

    var baseAnalyticsObject = JSON.parse(
      elem_.getAttribute('data-analytics').replace(/,(?=[^,]*$)/, '')
    );

    function getScrubbedUrl() {
      const urlParams = location.href.split('?');
      const hostPathUrl = urlParams[0];
      if(urlParams.length === 1) return hostPathUrl;

      try {
        const searchParams = urlParams[1].split('&');
        const utmParams = searchParams.filter(param => param.startsWith('utm') || param.startsWith('query')).join('&');
        if(!utmParams.length) return hostPathUrl;
        return `${hostPathUrl}?${utmParams}`;
      }
      catch(error) {
        console.error('scrubbed Url error =>', e);
      }
      return hostPathUrl;
    }

    function handlePublishDate(dataObj) {
      var publishDate = dataObj['publishDate'];

      if (typeof publishDate !== 'undefined') {
        var oneDay = 24*60*60*1000;
        var dateParts = publishDate.split('|');
        var pubDay = new Date(dateParts[0]);
        var today = new Date();
        var daysSince = Math.round((today - pubDay) / oneDay);

        dataObj['daysSincePublished'] = daysSince.toString();
      }

      pushData(dataObj);
    }

    function pushData(analyticsEvent) {
      if (!window['dataLayer']) {
        window['dataLayer'] = [];
      }
      window['dataLayer'].push(analyticsEvent);
      const globalKWDimenssions = Object.assign({}, analyticsEvent);
      if(globalKWDimenssions['event']) {
        delete globalKWDimenssions['event'];
      }
      window['globalKWCustomDimensions'] = globalKWDimenssions;
    }

    pushData(scrubObject);
    handlePublishDate(baseAnalyticsObject);
  }());
</script>

It was last updated on September 23, 2024.</p><h2 data-block-key="clbdp">April</h2><ul><li data-block-key="76gok">We terminated 1 Ads account and 1 AdSense account as part of our investigation into coordinated influence operations linked to Indonesia. The campaign was sharing content in Bahasa Indonesia that was supportive of the ruling party.</li><li data-block-key="27uph">We terminated 37 Blogger blogs as part of our investigation into coordinated influence operations linked to Indonesia. The campaign was sharing content in Bahasa Indonesia that was supportive of the ruling party.</li><li data-block-key="79qgo">We terminated 2 YouTube channels as part of our investigation into coordinated influence operations linked to Myanmar. The campaign was sharing content in Burmese and English that was supportive of the Burmese military government and critical of pro-independence groups.</li><li data-block-key="6e1e">We terminated 1 AdSense account and and blocked 10 domains from eligibility to appear on Google News surfaces and Discover as part of our investigation into coordinated influence operations linked to individuals in the Philippines and India. The campaign was sharing content in English and Norwegian about food, sports, and lifestyle topics. We believe this operation was financially motivated.</li><li data-block-key="9tfrj">We terminated 59 YouTube channels as part of our investigation into coordinated influence operations linked to Pakistan.The campaign was sharing Shorts content in Urdu that was critical of Pakistani political figures.</li><li data-block-key="feqg8">We terminated 11 YouTube channels as part of our investigation into coordinated influence operations linked to individuals in Russia. The campaign was sharing content supportive of Russia and critical of Ukraine.</li><li data-block-key="cmiiv">We terminated 378 YouTube channels as part of our investigation into coordinated influence operations linked to Russia. The campaign was linked to a Russian consulting firm and was sharing content in Russian that was supportive of Russia and critical of Ukraine and the West.</li><li data-block-key="boah5">We terminated 11 YouTube channels as part of our investigation into coordinated influence operations linked to individuals in France. The campaign was sharing content in French that was critical of French political figures.</li><li data-block-key="9jk2f">We terminated 1,320 YouTube channels and 1,177 Blogger blogs as part of our ongoing investigation into coordinated influence operations linked to the People’s Republic of China (PRC). These findings are consistent with our previous reports.</li><li data-block-key="8hsm9">We terminated 123 YouTube channels and 2 Ads accounts as part of our investigation into coordinated influence operations linked to India. The campaign was sharing content in Hindi and English.</li><li data-block-key="4lc0p">We terminated 88 YouTube channels as part of our investigation into coordinated influence operations linked to India. The campaign was linked to two digital marketing firms in India and was sharing content in Hindi that was supportive of an Indian politician in Maharashtra.</li><li data-block-key="31nm2">We terminated 13 YouTube channels as part of our investigation into coordinated influence operations linked to India. The campaign was sharing content in Hindi and English that was supportive of politicians in Rajasthan state.</li><li data-block-key="c6e42">We terminated 304 YouTube channels as part of our investigation into coordinated influence operations linked to India. The campaign was sharing content in Odia, Hindi and English that was supportive of politicians in Odisha state.</li><li data-block-key="figbm">We terminated 124 YouTube channels as part of our investigation into coordinated influence operations linked to India. We received leads from OpenAI that supported us in this investigation.</li><li data-block-key="9rbbu">We terminated 12 YouTube channels as part of our investigation into coordinated influence operations linked to Bangladesh. The campaign was sharing content in Bengali and English that was critical of the Bangladesh Nationalist Party (BNP). We received leads from Meta that supported us in this investigation.</li><li data-block-key="bb2r6">We terminated 4 YouTube channels as part of our investigation into coordinated influence operations linked to Iran. The campaign was sharing content in English, Farsi and Arabic that was supportive of Iran, Yemen, Palestine and Islam and critical of Israel.</li><li data-block-key="cllc3">We terminated 2 YouTube channels and blocked 2 domains from eligibility to appear on Google News surfaces and Discover as part of our investigation into coordinated influence operations linked to Israel. The campaign was linked to the Israeli company STOIC and was sharing content in English that was critical of Muslim immigration to Canada. We received leads from OpenAI and Meta that supported us in this investigation.</li><li data-block-key="3la4k">We terminated 1,595 YouTube channels and 32 Blogger blogs as part of our ongoing investigation into coordinated influence operations linked to the People’s Republic of China (PRC).
